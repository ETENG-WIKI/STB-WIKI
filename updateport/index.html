修改static-html-builder启动端口号

tasks\connect.js
options: {
  port: 9000 //启动端口
}, 
dev: {
      options: {
        livereload: true,//端口号
        middleware: require('../lib/middleware')
      }
    }

tasks\watch.js
 dev: {
      files: '<%= config.app %>/**/*.*',
      options: {
        livereload: true //端口号
      }
    }
